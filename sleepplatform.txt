// ══ SECURITY MODULE ══════════════════════════════════════
// Zero Trust: re-validate permissions on every sensitive operation
// AES-256-GCM encryption via WebCrypto API
// PBKDF2 key derivation (100,000 iterations, SHA-256)
// k-Anonymity: suppress stats when group < K_MIN
// Differential Privacy: Laplace noise on aggregate metrics

const SECURITY_VERSION = "v1.0";
const K_MIN = 5;       // k-anonymity threshold
const DP_EPSILON = 0.8; // differential privacy budget

// ── AES-256-GCM Crypto Engine ────────────────────────
const CryptoEngine = {
  _keyCache: new Map(),
  async deriveKey(orgCode, pin) {
    const cacheKey = orgCode + ":" + pin;
    if (this._keyCache.has(cacheKey)) return this._keyCache.get(cacheKey);
    try {
      const enc = new TextEncoder();
      const km = await window.crypto.subtle.importKey(
        "raw", enc.encode(pin + orgCode),
        {name:"PBKDF2"}, false, ["deriveKey"]
      );
      const key = await window.crypto.subtle.deriveKey(
        {name:"PBKDF2", salt:enc.encode("REIBI-SALT-2025-v1"),
         iterations:100000, hash:"SHA-256"},
        km, {name:"AES-GCM", length:256}, false, ["encrypt","decrypt"]
      );
      this._keyCache.set(cacheKey, key);
      return key;
    } catch { return null; }
  },
  async encrypt(key, plaintext) {
    if (!key) return null;
    try {
      const enc = new TextEncoder();
      const iv = window.crypto.getRandomValues(new Uint8Array(12));
      const ct = await window.crypto.subtle.encrypt(
        {name:"AES-GCM", iv}, key, enc.encode(plaintext)
      );
      // Store iv + ciphertext as base64
      const combined = new Uint8Array(iv.length + ct.byteLength);
      combined.set(iv); combined.set(new Uint8Array(ct), iv.length);
      return btoa(String.fromCharCode(...combined));
    } catch { return null; }
  },
  async decrypt(key, b64) {
    if (!key || !b64) return null;
    try {
      const bytes = Uint8Array.from(atob(b64), c => c.charCodeAt(0));
      const iv = bytes.slice(0, 12), ct = bytes.slice(12);
      const pt = await window.crypto.subtle.decrypt(
        {name:"AES-GCM", iv}, key, ct
      );
      return new TextDecoder().decode(pt);
    } catch { return null; }
  },
  clearKey(orgCode, pin) {
    this._keyCache.delete(orgCode + ":" + pin);
  }
};

// ── Differential Privacy (Laplace Mechanism) ─────────
const dpNoise = (sensitivity=1, epsilon=DP_EPSILON) => {
  const u = Math.random() - 0.5;
  return (sensitivity / epsilon) * Math.sign(u) * Math.log(1 - 2*Math.abs(u));
};
const dpCount = (trueCount) => Math.max(0, Math.round(trueCount + dpNoise(1)));
const dpPct = (truePct) => Math.min(100, Math.max(0, Math.round(truePct + dpNoise(5))));

// ── k-Anonymity Guard ─────────────────────────────────
const kAnonymize = (data, showFn) => {
  if (!Array.isArray(data) || data.length < K_MIN) return null;
  return showFn(data);
};

// ── Zero Trust Session Validator ──────────────────────
const ZeroTrust = {
  lastActivity: Date.now(),
  TIMEOUT_MS: 30 * 60 * 1000, // 30 min
  touch() { this.lastActivity = Date.now(); },
  isExpired() { return (Date.now() - this.lastActivity) > this.TIMEOUT_MS; },
  validateAction(session, permission) {
    if (!session) return false;
    if (this.isExpired()) return false;
    this.touch();
    const ok = can(session.systemRole, permission);
    AuditLog.record(
      ok ? "ACCESS_GRANTED" : "ACCESS_DENIED",
      permission, session.systemRole
    );
    return ok;
  }
};

// ── STORAGE (with optional AES-256-GCM encryption) ──
const stor={
  async get(k){
    try{
      const r=await Promise.race([
        window.storage.get(k),
        new Promise(res=>setTimeout(()=>res(null),2500))
      ]);
      if(!r?.value) return null;
      // Try decrypting if key available and value looks encrypted
      if(_cryptoKey && r.value.startsWith('"ENC:')) {
        const b64 = JSON.parse(r.value).slice(4);
        const pt = await CryptoEngine.decrypt(_cryptoKey, b64);
        return pt ? JSON.parse(pt) : null;
      }
      return JSON.parse(r.value);
    }catch{return null;}
  },
  async set(k,v){
    try{
      let payload;
      if(_cryptoKey) {
        const ct = await CryptoEngine.encrypt(_cryptoKey, JSON.stringify(v));
        payload = ct ? JSON.stringify("ENC:" + ct) : JSON.stringify(v);
      } else {
        payload = JSON.stringify(v);
      }
      await Promise.race([
        window.storage.set(k, payload),
        new Promise(res=>setTimeout(res,2500))
      ]);
    }catch{}
  },
  async del(k){try{await window.storage.delete(k);}catch{}}
};

// ══ SVG TREND CHART ══════════════════════════════════════
const TrendChart=({data,color,label,maxVal,height=90,showPred=false})=>{
  if(!data||data.length<2)return <div style={{textAlign:"center",padding:"20px",color:C.faint,fontSize:11}}>需至少2筆資料才能顯示趨勢圖</div>;
  const W=260,H=height,pad={t:10,r:12,b:28,l:30};
  const cW=W-pad.l-pad.r,cH=H-pad.t-pad.b;
  const mx=maxVal||Math.max(...data.map(d=>d.v))*1.15||1;
  const xs=data.map((_,i)=>pad.l+i/(data.length-1)*cW);
  const ys=data.map(d=>pad.t+cH*(1-d.v/mx));
  // Linear regression for prediction
  const n=data.length;
  const xm=(n-1)/2,ym=data.reduce((a,d)=>a+d.v,0)/n;
  const sxy=data.reduce((a,d,i)=>a+(i-xm)*(d.v-ym),0);
  const sxx=data.reduce((a,_,i)=>a+(i-xm)**2,0);
  const slope=sxx>0?sxy/sxx:0,intercept=ym-slope*xm;
  const pred=d=>Math.max(0,Math.min(mx,slope*(d)+intercept));
  const trend=data.map((_,i)=>({x:xs[i],y:pad.t+cH*(1-pred(i)/mx)}));
  const predX=pad.l+(n/(n-1))*cW,predY=pad.t+cH*(1-pred(n)/mx);
  const ptStr=xs.map((x,i)=>`${x},${ys[i]}`).join(" ");
  const trStr=trend.map(p=>`${p.x},${p.y}`).join(" ");
  const aStr=`${xs[0]},${ys[0]} ${ptStr} ${xs[xs.length-1]},${pad.t+cH} ${xs[0]},${pad.t+cH}`;
  return(
    <div>
      <svg width="100%" viewBox={`0 0 ${W} ${H}`} style={{overflow:"visible"}}>
        {/* Grid */}
        {[0,0.25,0.5,0.75,1].map(pct=>{const gy=pad.t+cH*pct;return(
          <g key={pct}>
            <line x1={pad.l} y1={gy} x2={pad.l+cW} y2={gy} stroke="#f0e8e0" strokeWidth="0.5"/>
            <text x={pad.l-3} y={gy+3} fontSize="7" fill="#b8afa8" textAnchor="end">{Math.round(mx*(1-pct))}</text>
          </g>
        );})}
        {/* Area fill */}
        <polygon points={aStr} fill={color} opacity="0.06"/>
        {/* Actual line */}
        <polyline points={ptStr} fill="none" stroke={color} strokeWidth="2" strokeLinejoin="round"/>
        {/* Trend line */}
        {showPred&&<polyline points={trStr} fill="none" stroke={color} strokeWidth="1" strokeDasharray="3,2" opacity="0.5"/>}
        {/* Predicted next point */}
        {showPred&&predY>0&&predY<H&&(
          <g>
            <line x1={xs[xs.length-1]} y1={ys[ys.length-1]} x2={predX} y2={predY} stroke={color} strokeWidth="1.5" strokeDasharray="3,2" opacity="0.7"/>
            <circle cx={predX} cy={predY} r="4" fill="white" stroke={color} strokeWidth="1.5" opacity="0.8"/>
            <text x={predX} y={predY-7} fontSize="8" fill={color} textAnchor="middle" fontWeight="600">預測</text>
          </g>
        )}
        {/* Data points */}
        {xs.map((x,i)=>(
          <g key={i}>
            <circle cx={x} cy={ys[i]} r="3.5" fill={color} stroke="white" strokeWidth="1.5"/>
            <text x={x} y={ys[i]-7} fontSize="7.5" fill={color} textAnchor="middle" fontWeight="600">{data[i].v}</text>
          </g>
        ))}
        {/* X axis labels */}
        {data.map((d,i)=>(
          <text key={i} x={xs[i]} y={H-5} fontSize="7" fill="#b8afa8" textAnchor="middle">{d.label}</text>
        ))}
        {/* Y axis label */}
        <text x={pad.l-18} y={pad.t+cH/2} fontSize="8" fill={color} textAnchor="middle" transform={`rotate(-90,${pad.l-18},${pad.t+cH/2})`}>{label}</text>
      </svg>
      {showPred&&<div style={{fontSize:9,color:C.muted,textAlign:"right",marginTop:2}}>虛線=趨勢推算（非醫療預測）</div>}
    </div>
  );
};

      });
    }
  },[screen,pending]);

  const login=useCallback(async s=>{setSession(s);sessionRef.current=s;ZeroTrust.touch();AuditLog.record('LOGIN','login',s.systemRole);const r=await DB.loadReports();setHistory(Array.isArray(r)?r:[]);setScreen("dash");},[]);
  const logout=useCallback(async()=>{await DB.clearSession();clearCryptoKey();AuditLog.record('LOGOUT','user logged out',session?.systemRole||'');setSession(null);sessionRef.current=null;setScreen('login');setHistory([]);},[session]);
  const go=s=>setScreen(s);
  const back=()=>{const m={profile:"dash",sleep:"profile",pain:"sleep",work:"pain",report:"dash",history:"dash",org:"dash",okr:"dash",appt:"dash",esg:"dash",analysis:"history",highrisk:"dash",privacy:"dash"};setScreen(m[screen]||"dash");};
  const LABEL={boot:"",login:"登入",dash:"主選單",profile:"個人資料",sleep:"🌙 睡眠評估",pain:"🩺 疼痛評估",work:"💼 工作評估",loading:"⚙️ 生成中",report:"📋 評估報告",history:"📈 歷史追蹤",org:"📊 KPI報表",okr:"🎯 OKR儀表板",appt:"📅 預約排程",esg:"🌱 ESG報告",analysis:"📊 健康分析",highrisk:"🔴 高風險分析",privacy:"🔒 隱私安全中心"};
  if(screen==="boot")return <div style={{minHeight:"100vh",display:"flex",alignItems:"center",justifyContent:"center",background:C.bg}}><div style={{fontSize:44,animation:"spin 2s linear infinite"}}>🌿</div><style>{`@keyframes spin{from{transform:rotate(0)}to{transform:rotate(360deg)}}`}</style></div>;
  return (
    <div style={{minHeight:"100vh",background:C.bg,fontFamily:"'Noto Serif TC','Microsoft JhengHei','Georgia',serif",padding:"20px 13px 60px",color:C.text}}>
      <div style={{maxWidth:680,margin:"0 auto"}}>
        <div style={{display:"flex",alignItems:"center",justifyContent:"space-between",marginBottom:16,paddingBottom:10,borderBottom:`1px solid ${C.divider}`}}>
          <div style={{display:"flex",alignItems:"center",gap:9}}><span style={{fontSize:20}}>🌿</span><div><div style={{fontSize:12,fontWeight:700,color:C.text}}>健康自主管理評估系統</div><div style={{fontSize:9,color:C.faint}}>Sleep × Pain · REIBI Health Platform</div></div></div>
          <div style={{display:"flex",alignItems:"center",gap:7}}>
            {session&&<span style={{display:"inline-flex",alignItems:"center",gap:4,padding:"3px 10px",borderRadius:20,background:`${ROLES[session.systemRole]?.color||C.teal}18`,border:`1px solid ${ROLES[session.systemRole]?.color||C.teal}44`,fontSize:11,color:ROLES[session.systemRole]?.color||C.teal,fontWeight:600}}>{ROLES[session.systemRole]?.icon} {ROLES[session.systemRole]?.label}</span>}
            {screen!=="login"&&screen!=="dash"&&screen!=="boot"&&<div style={{fontSize:9,color:C.teal,background:C.tealBg,border:`1px solid ${C.tealLight}`,borderRadius:20,padding:"2px 9px"}}>{LABEL[screen]}</div>}
          </div>
        </div>
        {showDecl&&<DeclModal onAgree={()=>{setDeclTs(new Date().toISOString());setShowDecl(false);setSAns({});setPAns({});setWAns({});setScreen("profile");}} onCancel={()=>setShowDecl(false)}/>}
        {screen==="login"&&<LoginScreen onLogin={login}/>}
        {screen==="dash"&&session&&<Dashboard session={session} history={history} onNav={a=>{if(a==="assess")setShowDecl(true);else go(a);}} onLogout={logout}/>}
        {screen==="profile"&&<ProfileScreen session={session} onBack={back} onNext={p=>{setProfile(p);go("sleep");}}/>}
        {screen==="sleep"&&<SleepQ ans={sAns} setAns={setSAns} onBack={back} onNext={()=>go("pain")}/>}
        {screen==="pain"&&<PainQ ans={pAns} setAns={setPAns} onBack={back} onNext={()=>go("work")}/>}
        {screen==="work"&&<WorkQ ans={wAns} setAns={setWAns} onBack={back} onNext={()=>{const snap=sessionRef.current||session;setPending({sa:{...sAns},pa:{...pAns},wa:{...wAns},sess:snap});go("loading");}}/>}
        {screen==="loading"&&<LoadingScreen/>}
        {screen==="report"&&report&&<ReportScreen report={report} session={session} onBack={back} onHistory={()=>{DB.loadReports().then(r=>{setHistory(Array.isArray(r)?r:[]);go("history");});}} onAnalysis={()=>go("analysis")}/>}
        {screen==="history"&&<HistoryScreen reports={history} session={session} onBack={back} onNew={()=>setShowDecl(true)} onAnalysis={()=>go("analysis")}/>}
        {screen==="org"&&session&&<OrgKPIScreen session={session} onBack={back}/>}
        {screen==="okr"&&session&&<OKRScreen session={session} onBack={back}/>}
        {screen==="appt"&&session&&<ApptScreen session={session} onBack={back}/>}
        {screen==="esg"&&session&&<ESGScreen session={session} onBack={back}/>}
        {screen==="analysis"&&session&&<AnalysisPage reports={history} session={session} onBack={back} onAppt={()=>go("appt")}/>}
        {screen==="highrisk"&&session&&<HighRiskWrapper session={session} onBack={back}/>}
        {screen==="privacy"&&session&&<PrivacyCenter session={session} onBack={back} onDataErase={async()=>{await DB.clearSession();clearCryptoKey();AuditLog.record("DATA_ERASE","user requested data erasure",session?.systemRole);setSession(null);setHistory([]);setScreen("login");}}/>}
      </div>
    </div>
  );
}
